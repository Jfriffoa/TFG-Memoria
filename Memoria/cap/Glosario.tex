\makeglossaries

\newglossaryentry{ide}{
	name=IDE,
	description={Ambiente de desarrollo de trabajo. Es un programa que apoya y facilita la labor de un programador}
}

\newglossaryentry{engine}{
    name=game engine,
    description={Programa que proporciona herramientas y facilita la creación de un videojuego}
}

\newglossaryentry{unity}{
    name=Unity3D,
    description={Engine para el desarrollo de videojuegos en 2d y 3d con posibilidad de exportar a diversas plataformas}
}

\newglossaryentry{smartphone}{
    name=smartphone,
    description={Teléfono inteligente, comúnmente un dispositivo Android o iPhone}
}

\newglossaryentry{moral-foundation}{
    name=Moral Foundations Questionnaire,
    description={Teoría creada por un grupo de psicólogos sociales y culturales para entender la moralidad}
}

\newglossaryentry{firebase}{
    name=Firebase,
    description={Plataforma de Backend desarrollada y alojada por Google}
}

\newglossaryentry{apk}{
    name=apk,
    description={Formato de paquetes para la instalación de programas en dispositivos Android}
}

\newglossaryentry{drive}{
    name=Google Drive,
    description={Sistema de Almacenamiento en la nube alojado por Google}
}

\newglossaryentry{sources}{
    name=sources,
    description={Elementos de la economía del juego que generan ``moneda''}
}

\newglossaryentry{drains}{
    name=drains,
    description={Elementos de la economía del juego que consumen la ``moneda''}
}

\newglossaryentry{converters}{
    name=converters,
    description={Elementos de la economía del juego que transforman la ``moneda'' en otro elemento (u otro elemento en la ``moneda'')}
}

\newglossaryentry{traders}{
    name=traders,
    description={Elementos de la economía del juego que traspasan la ``moneda'' a otro jugador o ente del juego}
}

\newglossaryentry{backend}{
    name=backend,
    description={Parte de un sistema informático que procesa datos y realiza funciones que el usuario final no ve directamente.}
}
